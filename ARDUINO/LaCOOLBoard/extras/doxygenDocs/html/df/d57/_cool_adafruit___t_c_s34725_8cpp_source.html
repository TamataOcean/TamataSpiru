271 uint16_t Adafruit_TCS34725::calculateColorTemperature(uint16_t r, uint16_t g, uint16_t b)
272 {
273  float X, Y, Z; /* RGB to XYZ correlation */
274  float xc, yc; /* Chromaticity co-ordinates */
275  float n; /* McCamy's formula */
276  float cct;
277 
278  /* 1. Map RGB values to their XYZ counterparts. */
279  /* Based on 6500K fluorescent, 3000K fluorescent */
280  /* and 60W incandescent values for a wide range. */
281  /* Note: Y = Illuminance or lux */
282  X = (-0.14282F * r) + (1.54924F * g) + (-0.95641F * b);
283  Y = (-0.32466F * r) + (1.57837F * g) + (-0.73191F * b);
284  Z = (-0.68202F * r) + (0.77073F * g) + ( 0.56332F * b);
285 
286  /* 2. Calculate the chromaticity co-ordinates */
287  xc = (X) / (X + Y + Z);
288  yc = (Y) / (X + Y + Z);
289 
290  /* 3. Use McCamy's formula to determine the CCT */
291  n = (xc - 0.3320F) / (0.1858F - yc);
292 
293  /* Calculate the final CCT */
294  cct = (449.0F * powf(n, 3)) + (3525.0F * powf(n, 2)) + (6823.3F * n) + 5520.33F;
295 
296  /* Return the results in degrees Kelvin */
297  return (uint16_t)cct;
298 }
