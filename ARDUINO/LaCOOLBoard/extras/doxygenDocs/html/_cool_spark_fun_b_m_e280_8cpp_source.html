69 uint8_t BME280::begin()
70 {
71  //Check the settings structure values to determine how to setup the device
72  uint8_t dataToWrite = 0; //Temporary variable
73  switch (settings.commInterface)
74  {
75 
76  case I2C_MODE:
77  Wire.begin(2,14);
78  break;
79 
80  case SPI_MODE:
81  // start the SPI library:
82  SPI.begin();
83  #ifdef ARDUINO_ARCH_ESP32
84  SPI.setFrequency(1000000);
85  // Data is read and written MSb first.
86  SPI.setBitOrder(SPI_MSBFIRST);
87  // Like the standard arduino/teensy comment below, mode0 seems wrong according to standards
88  // but conforms to the timing diagrams when used for the ESP32
89  SPI.setDataMode(SPI_MODE0);
90  #else
91  // Maximum SPI frequency is 10MHz, could divide by 2 here:
92  SPI.setClockDivider(SPI_CLOCK_DIV32);
93  // Data is read and written MSb first.
94  SPI.setBitOrder(MSBFIRST);
95  // Data is captured on rising edge of clock (CPHA = 0)
96  // Base value of the clock is HIGH (CPOL = 1)
97  // This was SPI_MODE3 for RedBoard, but I had to change to
98  // MODE0 for Teensy 3.1 operation
99  SPI.setDataMode(SPI_MODE3);
100  #endif
101  // initalize the data ready and chip select pins:
102  pinMode(settings.chipSelectPin, OUTPUT);
103  digitalWrite(settings.chipSelectPin, HIGH);
104  break;
105 
106  default:
107  break;
108  }
109 
110  //Reading all compensation data, range 0x88:A1, 0xE1:E7
111 
112  calibration.dig_T1 = ((uint16_t)((readRegister(BME280_DIG_T1_MSB_REG) << 8) + readRegister(BME280_DIG_T1_LSB_REG)));
113  calibration.dig_T2 = ((int16_t)((readRegister(BME280_DIG_T2_MSB_REG) << 8) + readRegister(BME280_DIG_T2_LSB_REG)));
114  calibration.dig_T3 = ((int16_t)((readRegister(BME280_DIG_T3_MSB_REG) << 8) + readRegister(BME280_DIG_T3_LSB_REG)));
115 
116  calibration.dig_P1 = ((uint16_t)((readRegister(BME280_DIG_P1_MSB_REG) << 8) + readRegister(BME280_DIG_P1_LSB_REG)));
117  calibration.dig_P2 = ((int16_t)((readRegister(BME280_DIG_P2_MSB_REG) << 8) + readRegister(BME280_DIG_P2_LSB_REG)));
118  calibration.dig_P3 = ((int16_t)((readRegister(BME280_DIG_P3_MSB_REG) << 8) + readRegister(BME280_DIG_P3_LSB_REG)));
119  calibration.dig_P4 = ((int16_t)((readRegister(BME280_DIG_P4_MSB_REG) << 8) + readRegister(BME280_DIG_P4_LSB_REG)));
120  calibration.dig_P5 = ((int16_t)((readRegister(BME280_DIG_P5_MSB_REG) << 8) + readRegister(BME280_DIG_P5_LSB_REG)));
121  calibration.dig_P6 = ((int16_t)((readRegister(BME280_DIG_P6_MSB_REG) << 8) + readRegister(BME280_DIG_P6_LSB_REG)));
122  calibration.dig_P7 = ((int16_t)((readRegister(BME280_DIG_P7_MSB_REG) << 8) + readRegister(BME280_DIG_P7_LSB_REG)));
123  calibration.dig_P8 = ((int16_t)((readRegister(BME280_DIG_P8_MSB_REG) << 8) + readRegister(BME280_DIG_P8_LSB_REG)));
124  calibration.dig_P9 = ((int16_t)((readRegister(BME280_DIG_P9_MSB_REG) << 8) + readRegister(BME280_DIG_P9_LSB_REG)));
125 
126  calibration.dig_H1 = ((uint8_t)(readRegister(BME280_DIG_H1_REG)));
127  calibration.dig_H2 = ((int16_t)((readRegister(BME280_DIG_H2_MSB_REG) << 8) + readRegister(BME280_DIG_H2_LSB_REG)));
128  calibration.dig_H3 = ((uint8_t)(readRegister(BME280_DIG_H3_REG)));
129  calibration.dig_H4 = ((int16_t)((readRegister(BME280_DIG_H4_MSB_REG) << 4) + (readRegister(BME280_DIG_H4_LSB_REG) & 0x0F)));
130  calibration.dig_H5 = ((int16_t)((readRegister(BME280_DIG_H5_MSB_REG) << 4) + ((readRegister(BME280_DIG_H4_LSB_REG) >> 4) & 0x0F)));
131  calibration.dig_H6 = ((uint8_t)readRegister(BME280_DIG_H6_REG));
132 
133  //Set the oversampling control words.
134  //config will only be writeable in sleep mode, so first insure that.
135  writeRegister(BME280_CTRL_MEAS_REG, 0x00);
136 
137  //Set the config word
138  dataToWrite = (settings.tStandby << 0x5) & 0xE0;
139  dataToWrite |= (settings.filter << 0x02) & 0x1C;
140  writeRegister(BME280_CONFIG_REG, dataToWrite);
141 
142  //Set ctrl_hum first, then ctrl_meas to activate ctrl_hum
143  dataToWrite = settings.humidOverSample & 0x07; //all other bits can be ignored
144  writeRegister(BME280_CTRL_HUMIDITY_REG, dataToWrite);
145 
146  //set ctrl_meas
147  //First, set temp oversampling
148  dataToWrite = (settings.tempOverSample << 0x5) & 0xE0;
149  //Next, pressure oversampling
150  dataToWrite |= (settings.pressOverSample << 0x02) & 0x1C;
151  //Last, set mode
152  dataToWrite |= (settings.runMode) & 0x03;
153  //Load the byte
154  writeRegister(BME280_CTRL_MEAS_REG, dataToWrite);
155 
156  return readRegister(0xD0);

171 float BME280::readFloatPressure( void )
172 {
173 
174  // Returns pressure in Pa as unsigned 32 bit integer in Q24.8 format (24 integer bits and 8 fractional bits).
175  // Output value of “24674867” represents 24674867/256 = 96386.2 Pa = 963.862 hPa
176  uint8_t buffer[3];
177  readRegisterRegion(buffer, BME280_PRESSURE_MSB_REG, 3);
178  int32_t adc_P = ((uint32_t)buffer[0] << 12) | ((uint32_t)buffer[1] << 4) | ((buffer[2] >> 4) & 0x0F);
179 
180  int64_t var1, var2, p_acc;
181  var1 = ((int64_t)t_fine) - 128000;
182  var2 = var1 * var1 * (int64_t)calibration.dig_P6;
183  var2 = var2 + ((var1 * (int64_t)calibration.dig_P5)<<17);
184  var2 = var2 + (((int64_t)calibration.dig_P4)<<35);
185  var1 = ((var1 * var1 * (int64_t)calibration.dig_P3)>>8) + ((var1 * (int64_t)calibration.dig_P2)<<12);
186  var1 = (((((int64_t)1)<<47)+var1))*((int64_t)calibration.dig_P1)>>33;
187  if (var1 == 0)
188  {
189  return 0; // avoid exception caused by division by zero
190  }
191  p_acc = 1048576 - adc_P;
192  p_acc = (((p_acc<<31) - var2)*3125)/var1;
193  var1 = (((int64_t)calibration.dig_P9) * (p_acc>>13) * (p_acc>>13)) >> 25;
194  var2 = (((int64_t)calibration.dig_P8) * p_acc) >> 19;
195  p_acc = ((p_acc + var1 + var2) >> 8) + (((int64_t)calibration.dig_P7)<<4);
196 
197  return (float)p_acc / 256.0;
198 
199 }

254 float BME280::readTempC( void )
255 {
256  // Returns temperature in DegC, resolution is 0.01 DegC. Output value of “5123” equals 51.23 DegC.
257  // t_fine carries fine temperature as global value
258 
259  //get the reading (adc_T);
260  uint8_t buffer[3];
261  readRegisterRegion(buffer, BME280_TEMPERATURE_MSB_REG, 3);
262  int32_t adc_T = ((uint32_t)buffer[0] << 12) | ((uint32_t)buffer[1] << 4) | ((buffer[2] >> 4) & 0x0F);
263 
264  //By datasheet, calibrate
265  int64_t var1, var2;
266 
267  var1 = ((((adc_T>>3) - ((int32_t)calibration.dig_T1<<1))) * ((int32_t)calibration.dig_T2)) >> 11;
268  var2 = (((((adc_T>>4) - ((int32_t)calibration.dig_T1)) * ((adc_T>>4) - ((int32_t)calibration.dig_T1))) >> 12) *
269  ((int32_t)calibration.dig_T3)) >> 14;
270  t_fine = var1 + var2;
271  float output = (t_fine * 5 + 128) >> 8;
272 
273  output = output / 100;
274 
275  return output;

291 void BME280::readRegisterRegion(uint8_t *outputPointer , uint8_t offset, uint8_t length)
292 {
293  //define pointer that will point to the external space
294  uint8_t i = 0;
295  char c = 0;
296 
297  switch (settings.commInterface)
298  {
299 
300  case I2C_MODE:
301  Wire.beginTransmission(settings.I2CAddress);
302  Wire.write(offset);
303  Wire.endTransmission();
304 
305  // request bytes from slave device
306  Wire.requestFrom(settings.I2CAddress, length);
307  while ( (Wire.available()) && (i < length)) // slave may send less than requested
308  {
309  c = Wire.read(); // receive a byte as character
310  *outputPointer = c;
311  outputPointer++;
312  i++;
313  }
314  break;
315 
316  case SPI_MODE:
317  // take the chip select low to select the device:
318  digitalWrite(settings.chipSelectPin, LOW);
319  // send the device the register you want to read:
320  SPI.transfer(offset | 0x80); //Ored with "read request" bit
321  while ( i < length ) // slave may send less than requested
322  {
323  c = SPI.transfer(0x00); // receive a byte as character
324  *outputPointer = c;
325  outputPointer++;
326  i++;
327  }
328  // take the chip select high to de-select:
329  digitalWrite(settings.chipSelectPin, HIGH);
330  break;
331 
332  default:
333  break;
334  }
335 
336 }

338 uint8_t BME280::readRegister(uint8_t offset)
339 {

343  switch (settings.commInterface) {
344 
345  case I2C_MODE:
346  Wire.beginTransmission(settings.I2CAddress);
347  Wire.write(offset);
348  Wire.endTransmission();
349 
350  Wire.requestFrom(settings.I2CAddress, numBytes);
351  while ( Wire.available() ) // slave may send less than requested
352  {
353  result = Wire.read(); // receive a byte as a proper uint8_t
354  }
355  break;
356 
357  case SPI_MODE:
358  // take the chip select low to select the device:
359  digitalWrite(settings.chipSelectPin, LOW);
360  // send the device the register you want to read:
361  SPI.transfer(offset | 0x80); //Ored with "read request" bit
362  // send a value of 0 to read the first byte returned:
363  result = SPI.transfer(0x00);
364  // take the chip select high to de-select:
365  digitalWrite(settings.chipSelectPin, HIGH);
366  break;
367 
368  default:
369  break;
370  }

383 void BME280::writeRegister(uint8_t offset, uint8_t dataToWrite)
384 {
385  switch (settings.commInterface)
386  {
387  case I2C_MODE:
388  //Write the byte
389  Wire.beginTransmission(settings.I2CAddress);
390  Wire.write(offset);
391  Wire.write(dataToWrite);
392  Wire.endTransmission();
393  break;
394 
395  case SPI_MODE:
396  // take the chip select low to select the device:
397  digitalWrite(settings.chipSelectPin, LOW);
398  // send the device the register you want to read:
399  SPI.transfer(offset & 0x7F);
400  // send a value of 0 to read the first byte returned:
401  SPI.transfer(dataToWrite);
402  // decrement the number of bytes left to read:
403  // take the chip select high to de-select:
404  digitalWrite(settings.chipSelectPin, HIGH);
405  break;
406 
407  default:
408  break;
